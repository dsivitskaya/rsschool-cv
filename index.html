function isLeapYear(year) {
    const divisibleBy4 = year % 4 === 0;
    const divisibleBy100 = year % 100 === 0;
    const divisibleBy400 = year % 400 === 0;
    const notDivisibleBy100 = divisibleBy100 === false;
    const part1 = divisibleBy4 && notDivisibleBy100;

}
